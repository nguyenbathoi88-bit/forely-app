const express = require('express'); const Parser = require('rss-parser'); const axios = require('axios'); const app = express(); app.use(express.json()); // Khởi tạo thư viện đọc RSS cho tính năng Oracle const parser = new Parser(); /** * 1. CỔNG NHẬN WEBHOOK TỪ SEPAY / CASSO (AUTO CASH-IN) * Mục đích: Tự động cộng F-Point ngay khi người dùng chuyển khoản VietQR mua gói nội dung. */ app.post('/api/v1/webhooks/bank', async (req, res) => { try { const transaction = req.body; // Trích xuất nội dung chuyển khoản và số tiền từ payload của webhook const transferContent = transaction.content; // VD: "FORELY 8X9A2" const transferAmount = transaction.amount; // Logic: Tìm mã người dùng dựa trên nội dung chuyển khoản const userId = extractUserIdFromContent(transferContent); if (userId) { // Quy đổi số tiền nạp sang F-Point (VD: 10.000 VNĐ = 10.000 F-Point) const fPointsToAdd = calculateFPoints(transferAmount); // Cập nhật sổ cái Supabase (Giao dịch ACID) await supabase.rpc('add_fpoint_balance', { user_id: userId, points: fPointsToAdd }); console.log(`Đã cộng ${fPointsToAdd} F-Point cho user ${userId}`); } // Trả về 200 OK để hệ thống báo nhận thành công res.status(200).json({ success: true, message: "Webhook processed" }); } catch (error) { res.status(500).json({ error: "Internal Server Error" }); } }); /** * 2. CỔNG ORACLE CẬP NHẬT TIN TỨC VÀ KẾT QUẢ DỰ BÁO * Mục đích: Biến ứng dụng thành một Trang thông tin điện tử tổng hợp hợp pháp. */ app.get('/api/v1/news/fetch-oracle', async (req, res) => { try { // Lấy dữ liệu RSS từ các chuyên mục của báo điện tử (VD: Kinh tế VnExpress) const feedUrl = 'https://vnexpress.net/rss/kinh-te.rss'; const feed = await parser.parseURL(feedUrl); const newsList = feed.items.map(item => ({ title: item.title, link: item.link, pubDate: item.pubDate, snippet: item.contentSnippet })); // Lưu bản tin vào Database để hiển thị Frontend // Các câu hỏi dự báo (khảo sát) sẽ được đính kèm vào các bài báo này await supabase.from('news_articles').upsert(newsList); res.status(200).json({ success: true, message: "Đã đồng bộ tin tức Oracle thành công", data: newsList.length }); } catch (error) { res.status(500).json({ error: "Lỗi kết nối Oracle" }); } }); /** * 3. CỔNG OFF-RAMP: ĐỔI F-POINT LẤY VOUCHER QUA API * Mục đích: Tặng thưởng cho khách hàng thông qua đối tác kho quà thay vì tiền mặt. */ app.post('/api/v1/rewards/redeem', async (req, res) => { const { userId, voucherSku, requiredPoints } = req.body; try { // Bước 1: Trừ điểm F-Point trong Database trước (Đảm bảo an toàn giao dịch) const deduction = await supabase.rpc('deduct_fpoint_balance', { user_id: userId, points: requiredPoints }); if (!deduction.success) { return res.status(400).json({ error: "Số dư F-Point không đủ" }); } // Bước 2: Gọi API sang nền tảng đối tác kho quà (VD: UrBox) để xuất mã Voucher const urboxApiUrl = "https://api.urbox.vn/v1/gift/issue"; // Giả lập Endpoint const urboxResponse = await axios.post(urboxApiUrl, { sku: voucherSku, merchant_id: process.env.URBOX_MERCHANT_ID, secret_key: process.env.URBOX_SECRET_KEY }); const voucherCode = urboxResponse.data.voucher_code; // Bước 3: Trả mã Voucher về cho màn hình điện thoại người dùng res.status(200).json({ success: true, voucher: voucherCode, message: "Đổi thưởng thành công! Quà tặng của bạn đã sẵn sàng." }); } catch (error) { // Nếu API đối tác lỗi, tự động hoàn điểm (Auto-refund) lại cho người dùng await supabase.rpc('add_fpoint_balance', { user_id: userId, points: requiredPoints }); res.status(502).json({ error: "Lỗi cổng quà tặng, đã hoàn điểm cho bạn." }); } }); // Khởi động Server const PORT = process.env.PORT | | 4300; app.listen(PORT, () => { console.log(`Forely Backend Engine đang chạy trên cổng ${PORT}`); });
